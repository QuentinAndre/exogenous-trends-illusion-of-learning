generate_title <- function(model) {
  coeffs <- model %>% tidy()
  summ <- model %>% glance()
  df <- summ %>% slice(1) %>% pull(df.residual)
  t <- coeffs %>% slice(2) %>% pull(statistic)
  pval <- coeffs %>% slice(2) %>% pull(p.value)
  glue(
    "t({df}) = {round(t, 3)}, {ifelse(pval < .001, 
    'p < .001', glue('p = {round(pval, 3)}'))}"
  )
}
